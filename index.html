async function askAI() { const input = document.getElementById('userInput').value; const chatLog = document.getElementById('chatLog'); if(!input) return; chatLog.innerHTML += `

You: ${input}

`; chatLog.innerHTML += `

AI is thinking...

`; document.getElementById('userInput').value = ''; // IMPORTANT: Make sure your key is inside these quotes const API_KEY = "PASTE_YOUR_API_KEY_HERE"; try { const response = await fetch('https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=' + API_KEY, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ "contents": [{ "parts": [{ "text": input }] }] }) }); const data = await response.json(); document.getElementById('loading').remove(); if (data.candidates) { const aiReply = data.candidates[0].content.parts[0].text; chatLog.innerHTML += `

AI: ${aiReply}

`; } else { // This will tell us if there's an error message from Google chatLog.innerHTML += `

API Error: ${JSON.stringify(data)}

`; } } catch (error) { document.getElementById('loading').remove(); chatLog.innerHTML += `

Fetch Error: ${error.message}

`; } }
